import { ethers } from "ethers";
import { getSimpleAccount } from "./getSimpleAccount";

/**
 * Envia uma transa√ß√£o usando ERC-4337
 */
export async function sendTransaction(
  privateKey: string,
  to: string,
  value: string, // Em ETH (ex: "0.01")
  provider: any,
  config: any
) {
  try {
    console.log("üí∏ Iniciando transa√ß√£o ERC-4337...");
    
    // 1. Configurar Account API
    const accountAPI = getSimpleAccount(
      provider,
      privateKey,
      config.entryPoint,
      config.simpleAccountFactory
    );

    // 2. Criar UserOperation
    const userOp = await accountAPI.createSignedUserOp({
      target: to,
      data: "0x", // Transfer√™ncia simples sem dados
      value: ethers.utils.parseEther(value)
    });

    console.log("üìã UserOperation criada:", userOp);

    // 3. Enviar via Bundler
    const client = accountAPI.httpRpcClient;
    const userOpHash = await client.sendUserOpToBundler(userOp);
    
    console.log("üéÅ UserOp enviada ao bundler:", userOpHash);

    // 4. Aguardar confirma√ß√£o
    const receipt = await accountAPI.getUserOpReceipt(userOpHash);
    
    if (receipt) {
      console.log("‚úÖ Transa√ß√£o confirmada:", receipt);
      return { success: true, txHash: receipt };
    } else {
      throw new Error("Transa√ß√£o n√£o confirmada");
    }

  } catch (error) {
    console.error("‚ùå Erro na transa√ß√£o:", error);
    return { success: false, error };
  }
}

import { PaymasterAPI } from "@account-abstraction/sdk";

/**
 * Paymaster que patrocina gas para usu√°rios
 */
export class SponsoredPaymaster extends PaymasterAPI {
  private paymasterUrl: string;
  private sponsorApiKey: string;

  constructor(paymasterUrl: string, sponsorApiKey: string) {
    super();
    this.paymasterUrl = paymasterUrl;
    this.sponsorApiKey = sponsorApiKey;
  }

  async getPaymasterAndData(userOp: any): Promise<string> {
    try {
      // Chama API do paymaster para obter dados de patroc√≠nio
      const response = await fetch(`${this.paymasterUrl}/sponsor`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${this.sponsorApiKey}`
        },
        body: JSON.stringify({
          userOperation: userOp,
          entryPoint: "0x1306b01bC3e4AD202612D3843387e94737673F53"
        })
      });

      const data = await response.json();
      
      if (data.paymasterAndData) {
        console.log("üí≥ Gas patrocinado pelo paymaster!");
        return data.paymasterAndData;
      }

      return "0x"; // Sem patroc√≠nio, usu√°rio paga
    } catch (error) {
      console.log("üí∏ Paymaster indispon√≠vel, usu√°rio pagar√° gas");
      return "0x";
    }
  }
}

// Uso no getSimpleAccount.ts
export function getSimpleAccountWithPaymaster(
  provider: any,
  signingKey: string,
  entryPointAddress: string,
  factoryAddress: string,
  paymasterUrl?: string,

) {

import { ethers } from "ethers";

/**
 * Executa m√∫ltiplas opera√ß√µes em uma √∫nica UserOperation
 */
export async function executeBatch(
  accountAPI: any,
  operations: Array<{
    to: string;
    value?: string;
    data?: string;
    description: string;
  }>
) {
  console.log(`‚ö° Executando ${operations.length} opera√ß√µes em lote:`);
  operations.forEach((op, i) => {
    console.log(`  ${i + 1}. ${op.description}`);
  });

  try {
    // Preparar calls para execu√ß√£o em lote
    const calls = operations.map(op => ({
      target: op.to,
      value: op.value ? ethers.utils.parseEther(op.value) : 0,
      data: op.data || "0x"
    }));

    // Criar UserOperation para batch
    const userOp = await accountAPI.createSignedUserOp({
      target: accountAPI.getAccountAddress(), // Chama a pr√≥pria smart wallet
      data: accountAPI.encodeBatch(calls)      // Encode das m√∫ltiplas opera√ß√µes
    });

    // Enviar
    const userOpHash = await accountAPI.httpRpcClient.sendUserOpToBundler(userOp);
    console.log("üéÅ Batch enviado:", userOpHash);

    // Aguardar confirma√ß√£o
    const receipt = await accountAPI.getUserOpReceipt(userOpHash);
    
    if (receipt) {
      console.log("‚úÖ Todas as opera√ß√µes executadas com sucesso!");
      return { success: true, txHash: receipt };
    }

    throw new Error("Batch n√£o confirmado");

  } catch (error) {
    console.error("‚ùå Erro no batch:", error);
    return { success: false, error };
  }
}
